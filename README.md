# workspace
Personal Workspace

# Linux 
`
git clone git@github.com:enricogolfieri/workspace ~/.config/workspace 
`

update .bashrc

`
source ./config/workspace/linux/.bashrc
`

`

`
brew install miniconda 
`

conda install python=3.9 
`

update .zshrc
`
source ./config/workspace/mac/.zshrc
`
